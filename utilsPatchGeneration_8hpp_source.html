112// Hash function for vector3
113template <typename T>
114struct vector3Hash {
115 size_t operator()(const Vector3<T>& vector) const {
116 std::hash<T> hasher;
117 size_t hash = 0;
118 for (const auto& elem : vector) {
119 // Use a simple hash combining algorithm
120 hash ^= hasher(elem) + 0x9e3779b9 + (hash << 6U) + (hash >> 2U);
121 }
122 return hash;
123 }
124};

127// The voxelization is done so to have those three vectors sharing the same order (that is sharing the same voxel index) :
128// -> A list of voxel (vector of coordinates)
129// -> A list of list of points (vector of vector of point index). This associate a voxel with the points inside it.
130// -> A list of voxel PPI (vector of PPI (int))
131//
132// Example with a voxel index i:
133// voxelizedPointsGeometry[i] stores the coordinate of the voxel i
134// voxelIdToPointsId[i] stores the index of the points inside the voxel i
135// voxelsPPIs[i] stores the PPI of the voxel i
136//
137// The use of a map is needed as several points can be inside one voxel. We want to find in a fast way if the current point resides in a voxel
138// that has already been created. For an input point j: voxelCoordToVoxelIndex[ voxelCoordsOf(j) ] = i
139//
140// This structure makes computation easier for the patch generation and for applying voxel data to points data.
141// To achieve this, we need a temporary map that interfaces the voxel coordinates with the voxel index in those three list.
142
143// voxelizedPointsGeometry -> list of the voxels. That is, a list of the coordinates of the voxels.
144// voxelIdToPointsId -> For each voxel, stores the index of the points inside. A voxel is defined by a voxel index, that is given by the
145// filling order of the vector. Thus, the third voxel to be added to this structure will have a voxel index of 2.
146// voxelCoordToVoxelIndex -> Temporary structure. Associate a voxel to its index in 'voxelIdToPointsId' (its voxel index). This structure is a
147// map whose key is a voxel coordinate and the value is a voxel index.
148//
149// For each point if the input point cloud :
150// Compute the coordinates of the voxel inside which the current point should be.
151// If those voxel coordinates are already in the map 'voxelCoordToVoxelIndex':
152// Add the current input point index to the list of points index of the found voxel.
153// That is, add the current input point index to the list of points of the voxel in 'voxelIdToPointsId'
154// If those voxel coordinates are not in the map 'voxelCoordToVoxelIndex':
155// Create a new item in the map 'voxelCoordToVoxelIndex'. This associates the voxel coordinates to its voxel index (the current size
156// of the list of voxel === the index of this voxel in the list of voxel). Add a new voxel to 'voxelIdToPointsId'. This associates the
157// voxel index (the current size of the list of voxel === the index of this voxel in the list of voxel) to an empty list of input
158// point index. Create a new voxel by adding the voxel coordinates in the voxel list 'voxelizedPointsGeometry'. Add the current input
159// point index to the list of points index of the new voxel. That is, add the current input point index to the list of points of the
160// voxel in 'voxelIdToPointsId'
161
162
163// Notice that in the refined segmentation, the inputPointsGeometry can be the voxelizedPointsGeometry. Indeed, when the voxelization is
164// activated, the refined segmentation is applying a second voxelization step. The created voxelized point cloud is then the result of two
165// voxelizations.
166inline void voxelization(
167 const std::vector<Vector3<typeGeometryInput>>& inputPointsGeometry,
168 std::vector<Vector3<typeGeometryInput>>& voxelizedPointsGeometry,
169 std::vector<size_t>& pointsIdToVoxelId,
170 const size_t inputBitResolution,
171 const size_t outputBitResolution
172)
173{
174 Logger::log<LogLevel::TRACE>("PATCH GENERATION",
175 "Voxelization from " + std::to_string(inputBitResolution) + " to " + std::to_string(outputBitResolution) + " bits of resolution.\n");
176 pointsIdToVoxelId.resize(inputPointsGeometry.size());
177 const typeGeometryInput shift = static_cast<typeGeometryInput>(inputBitResolution - outputBitResolution);
178 const size_t approximateVoxelCount = 1U << (outputBitResolution * 2U);
179 voxelizedPointsGeometry.reserve(approximateVoxelCount);
180
181 robin_hood::unordered_map<Vector3<typeGeometryInput>, size_t, vector3Hash<typeGeometryInput>> voxelCoordToVoxelIndex;
182 voxelCoordToVoxelIndex.reserve(approximateVoxelCount);
183 size_t voxelIndex = 0;
184 for (size_t inputPointIndex = 0; inputPointIndex < inputPointsGeometry.size(); ++inputPointIndex) {
185 const Vector3<typeGeometryInput> & inputPoint = inputPointsGeometry[inputPointIndex];
186
187 Vector3<typeGeometryInput> voxCoord{
188 static_cast<typeGeometryInput>(static_cast<uint32_t>(inputPoint[0]) >> shift),
189 static_cast<typeGeometryInput>(static_cast<uint32_t>(inputPoint[1]) >> shift),
190 static_cast<typeGeometryInput>(static_cast<uint32_t>(inputPoint[2]) >> shift)
191 };
192
193 auto [it, inserted] = voxelCoordToVoxelIndex.try_emplace(voxCoord, voxelIndex);
194 if (inserted) {
195 voxelizedPointsGeometry.emplace_back(voxCoord);
196 voxelIndex++;
197 }
198 pointsIdToVoxelId[inputPointIndex] = voxelCoordToVoxelIndex[voxCoord];
199 }
